typedef double(*func) (double, double);

double func_s(double x, double n) {
		 double r=1, sum = 1;
		for (int k = 1; k <= n; k++) {
			r *= x * x / k;
			sum += r*(2 * k - 1)*x*x;
		}
		return sum;
}

double func_y(double x, double n) {
	return (1 + 2 * x * x) * exp(x * x);
}

double func_fabs(double x, double n) {
	return fabs(func_y(x, n) - func_s(x, n));
}
